LINKED LIST TRAVERSAL CODE

#include<stdio.h>
#include<stdlib.h>
struct node{
    int data;
    struct node *next;
};
int main()
{
    struct node *head;
    struct node *second;
    struct node *third;

    head= (struct node*)malloc(sizeof(struct node));
    second= (struct node*)malloc(sizeof(struct node));
    third=(struct node*)malloc(sizeof(struct node));
    head->data=13;
    head->next=second;
    second->data=2749;
    second->next=third;
    third->data=3094;
    third->next=NULL;
    nodetraversal(head);
    return 0;
     
}
 void nodetraversal(struct node *ptr)
{
    while(ptr!=NULL)
    {
    printf("element : %d\n", ptr->data);
    ptr=ptr->next;
    }
}
